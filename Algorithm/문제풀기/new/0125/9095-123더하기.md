# 1,2,3 더하기

```C
int answer;
int cnt;

void sumIt(int sum) {
    
    if(sum > answer) {
        return;
    }
    else if (sum == answer) {
        cnt += 1;
        return;
    }
    
    for(int i = 1; i<=3; i++) {
        sumIt(sum + i);
    }
}

int main(int argc, const char * argv[]) {
    int n;
    cin>>n;
    
    for(int i= 0 ;i <n; i++) {
        cin>>answer;
        cnt = 0;
        sumIt(0); 
        cout<<cnt<<endl;
    }
    return 0;
}
```
